介绍 ​

axios 封装统一加解密，实现配置化开发请求

使用方式 ​

  • 安装依赖
pnpm install @xxx/HRequest
  • 代码配置示例
ts
import { HRequest } from "@xxx/HRequest";
import { ResultEnum } from "@/enums/httpEnum";
import { useUserStore } from "@/stores/modules/user";
import {
  showFullScreenLoading,
  tryHideFullScreenLoading,
} from "@/components/Loading/fullScreen";

export default new HRequest({
  baseURL: import.meta.env.VITE_API_URL as string,
  timeout: ResultEnum.TIMEOUT as number,
  encrypt: true,
  loading: true,
  signature: true,
  withCredentials: true // 跨域时候允许携带凭证

  interceptors: {
    requestSuccessFn: (config) => {
      config.loading ??= true;
      config.loading && showFullScreenLoading(); //开启loading
      const userStore = useUserStore();
      const token = userStore.token
        ? userStore.token
        : "226e21a0-f915-4974-8a0e-7539cb940f4a";
      if (config.headers && token) {
        config.headers.set("X-Auth-Token", token);
      }
      return config;
    },
    requestFailFn:err=>{
      console.log(err);
      return err;
    }
    responseSuccessFn: (res) => {
      tryHideFullScreenLoading();
      console.log(res);
      return res;
    },
    responseFailFn: (res) => {
      tryHideFullScreenLoading();
      console.log(res);
      return res;
    },
  },
});

  • 配置属性

interceptors 拦截器配置对象：

属性名类型描述是否必填默认值
requestSuccessFn(config: any) => InternalAxiosRequestConfig处理成功请求拦截器的函数。否undefined
requestFailFn(err: any) => any处理失败请求拦截器的函数。否undefined
responseSuccessFn(res: T) => T处理成功响应拦截器的函数。否undefined
responseFailFn(err: any) => any处理失败响应拦截器的函数。否undefined

newRequestConfig 配置项：

属性名类型描述是否必填默认值
loadingboolean表示是否启用加载状态。否
signatureboolean表示是否启用签名。否
encryptboolean表示是否启用加密。否
  • 加解密配置方法

sxy-login 包登录

vue
<template>
  <XsyLogin systemId="1" @login-success="onLogin" />
</template>
<script setup lang="ts">
import XsyLogin from "xsy-login";
import "@xxx/xsy-login/dist/style.css";
import { cryptoSingleton } from "@xxx/HRequest";

const onLogin = async (data?: any) => {
  /**
   * 会话绑定
   */
  await cryptoSingleton.resetInitialization();
  await cryptoSingleton.init(import.meta.env.VITE_API_URL as string); //填的是baseURL
  await cryptoSingleton.newSm4(
    import.meta.env.VITE_API_URL as string,
    data?.token
  );
};
</script>

单独配置

cryptoSingleton.newSm4(baseURL, token);

3.拦截器配置

ts
    this.instance.interceptors.request.use(
      (config: any) => {
        const { url, params, data, method } = config;
        //加密判断
        if (config?.encrypt) {
          if (
            config.method?.toLowerCase() === "post" &&
            config.data &&
            config.headers["Content-Type"] !== "multipart/form-data"
          ) {
            const { cipherText, combinedKeyIv } = CryptoHelper.sm4DoEncrypt(
              JSON.stringify(config.data)
            );
            config.headers["x-crypto-key"] = combinedKeyIv;
            config.data = { cipherText };
          }
        }
        //数字签名判断
        if (config?.signature && config?.headers["X-Auth-Token"]) {
          let formData: FormData | null = null;
          if (data instanceof FormData) {
            formData = data;
          }
          const signParams = CryptoHelper.signature(
            url,
            params,
            config.data,
            formData,
            method,
            config.headers["X-Auth-Token"]
          );
          Object.assign(config.headers, signParams);
        }
        if (config.headers && typeof config.headers.set === "function") {
          config.headers.set("cryptoToken", cryptoSingleton.getCryptoToken());
          config.headers.set(
            "clientPublicKey",
            cryptoSingleton.clientPublicKey()
          );
        }

        return config;
      }
